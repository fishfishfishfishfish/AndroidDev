def get_image_mat(data_path, pic_size):
    img = image.imdecode(open(data_path, 'rb').read())
    if pic_size[0]*pic_size[1] != 0:
        img = image.ForceResizeAug(pic_size)(img)
    return img.astype('float32').asnumpy()

def get_labels(str):
    y = []
    for s in str:
        if s == 'y':
            y.append(1)
        elif s == 'm':
            y.append(0.5)
        else:
            y.append(0)
    return y

迭代读取数据，标签¶

  • data_path: 指向Annotations和Images所在路径
  • batch_size: 批量大小
  • shuffle: 决定是否打乱顺序
  • transform: 对数据和标签做变换的函数，一般用于图片增强
  • for_show: 决定图片维度是用于plt.imshow与否
  • pic_size: 图片读取的大小

class DataLoader(object):
    def __init__(self, data_path, batch_size, shuffle, transform=None, for_show=False, pic_size=(512, 512)):
        self.data_path = data_path
        self.batch_size = batch_size
        self.shuffle = shuffle
        self.transform = transform
        self.for_show = for_show
        self.pic_size = pic_size

    def __iter__(self):

        raw_data = pandas.read_csv(self.data_path + "Annotations/skirt_length_labels.csv")
        n = len(raw_data)
        if self.shuffle:
            idx = np.arange(n)
            np.random.shuffle(idx)
            raw_data = raw_data.iloc[idx]

        for i in range(n//self.batch_size):
            x = []
            y = []
            for j in range(i*self.batch_size, (i+1)*self.batch_size):
                x.append(get_image_mat(self.data_path + raw_data.iloc[j, 0], self.pic_size))
                y.append(get_labels(raw_data.iloc[j, 2]))
                # print(raw_data.iloc[j, 0])
            x = nd.array(np.array(x))
            y = nd.array(y)
            if self.transform is not None:
                x,y = self.transform(x, y)
            if self.for_show == False:
                x = nd.transpose(x, (0, 3, 1, 2))

            yield (x, y)

    def __len__(self):
        raw_data = pandas.read_csv(self.data_path + "Annotations/skirt_length_labels.csv")
        return len(raw_data)//self.batch_size
  • for_show为真时，输出的图片矩阵用于plt.imshow，颜色通道维度为第1维
  • for_show为假时，输出的图片矩阵用于输入神经网络，颜色通道维度为第3维

def apply_aug_list(img, augs):
    for f in augs:
        img = f(img)
    return img

把增强函数用以所有样本

def get_transform(augs):
    def transform(data, label):
        # data: sample x height x width x channel
        # label: sample
        data = data.astype('float32')
        if augs is not None:
            # apply to each sample one-by-one and then stack
            data = nd.stack(*[
                apply_aug_list(d, augs) for d in data])
        return data, label.astype('float32')
    return transform
